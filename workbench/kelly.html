coin_toss_seq <- function(
    num_rounds = 1,
    heads_prob = 0.5,
    initial = 100,
    head_factor = 2,
    tails_factor = 0.5,
    min_bet = 1,
    max_payout = NA,
    betting_fraction = 1) {
  results <- c(initial)
  toss_results <- rbinom(num_rounds, size = 1, prob = heads_prob)

  for (res in toss_results) {
    cur <- results[length(results)]
    if (cur > min_bet && (is.na(max_payout) || cur < max_payout)) {
      if (res == 0) {
        # tails
        results <- c(results, cur * tails_factor)
      } else {
        # heads
        results <- c(results, cur * head_factor)
      }
    }
  }

  if (!is.na(max_payout)) {
    results[length(results)] <- min(results[length(results)], max_payout)
  }

  return(
    list(
      results = results,
      toss_results = toss_results,
      final = results[length(results)]
    )
  )
}
